ما هي الفهارس؟

الفهارس تحسن أداء الاستعلامات عن طريق تسريع البحث في الجداول.

إنشاء Index

CREATE INDEX IX_Students_Name ON الطلاب(الاسم); -- Index مركب CREATE INDEX IX_Students_Specialty_Age ON الطلاب(التخصص, العمر);

Unique Index

CREATE UNIQUE INDEX IX_Students_Email ON الطلاب(البريد_الإلكتروني);

حذف Index

DROP INDEX IX_Students_Name ON الطلاب;
