ما هي الفهارس المكانية؟

الفهارس المكانية (Spatial Indexes) هي هياكل بيانات خاصة تُستخدم لتسريع الاستعلامات الجغرافية. PostGIS يستخدم فهرس GIST (Generalized Search Tree) لتحسين أداء الاستعلامات المكانية.

إنشاء الفهارس المكانية

إنشاء فهرس مكاني
-- فهرس بسيط
CREATE INDEX idx_cities_location ON cities USING GIST(location);

-- فهرس مع خيارات
CREATE INDEX idx_regions_boundary ON regions 
USING GIST(boundary) 
WITH (fillfactor = 90);

-- فهرس على عدة أعمدة
CREATE INDEX idx_roads_geom_type ON roads 
USING GIST(geometry, road_type);

التحقق من الفهارس

عرض الفهارس
-- عرض جميع الفهارس
SELECT 
    tablename,
    indexname,
    indexdef
FROM pg_indexes
WHERE tablename = 'cities';

-- عرض معلومات الفهارس
SELECT 
    schemaname,
    tablename,
    indexname,
    indexdef
FROM pg_indexes
WHERE indexdef LIKE '%GIST%';

استخدام الفهارس في الاستعلامات

الاستعلامات التي تستخدم الفهارس
-- هذه الاستعلامات تستخدم الفهرس تلقائياً:
-- ST_Intersects
SELECT * FROM cities c, regions r
WHERE ST_Intersects(c.location, r.boundary);

-- ST_Contains
SELECT * FROM cities c, regions r
WHERE ST_Contains(r.boundary, c.location);

-- ST_DWithin
SELECT * FROM cities c1, cities c2
WHERE ST_DWithin(
    ST_Transform(c1.location, 3857),
    ST_Transform(c2.location, 3857),
    10000
);

صيانة الفهارس

صيانة الفهارس
-- إعادة بناء الفهرس
REINDEX INDEX idx_cities_location;

-- إعادة بناء جميع الفهارس في جدول
REINDEX TABLE cities;

-- تحديث الإحصائيات
VACUUM ANALYZE cities;

-- التحقق من حجم الفهرس
SELECT 
    pg_size_pretty(pg_relation_size('idx_cities_location')) AS index_size;

أفضل الممارسات

  • أنشئ فهرسًا على كل عمود جغرافي
  • استخدم VACUUM ANALYZE بعد إدراج بيانات كثيرة
  • راقب أداء الاستعلامات باستخدام EXPLAIN
  • أعد بناء الفهارس دوريًا
