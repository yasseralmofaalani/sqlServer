المحاضرة 11: الدوال المجمعة (Aggregate Functions)

مقدم الدورة: م. ياسر المفعلاني

  • الصفحة الرئيسية
  • المحتوى

COUNT - العد

SELECT COUNT(*) AS عدد_الطلاب FROM الطلاب; SELECT COUNT(التخصص) AS عدد_التخصصات FROM الطلاب;

SUM - المجموع

SELECT SUM(الراتب) AS مجموع_الرواتب FROM الموظفون;

AVG - المتوسط

SELECT AVG(الدرجة) AS متوسط_الدرجات FROM التسجيلات;

MAX و MIN

SELECT MAX(الدرجة) AS أعلى_درجة, MIN(الدرجة) AS أقل_درجة FROM التسجيلات;

GROUP BY

SELECT التخصص, COUNT(*) AS العدد FROM الطلاب GROUP BY التخصص;

HAVING

SELECT التخصص, COUNT(*) AS العدد FROM الطلاب GROUP BY التخصص HAVING COUNT(*) > 10;
