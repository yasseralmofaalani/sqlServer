دوال النصوص

SELECT LEN('مرحباً'); -- الطول SELECT UPPER('hello'); -- أحرف كبيرة SELECT LOWER('HELLO'); -- أحرف صغيرة SELECT SUBSTRING('مرحباً', 1, 3); -- جزء من النص

دوال التاريخ والوقت

SELECT GETDATE(); -- التاريخ والوقت الحالي SELECT YEAR(GETDATE()); -- السنة SELECT MONTH(GETDATE()); -- الشهر SELECT DAY(GETDATE()); -- اليوم

دالة مخصصة

CREATE FUNCTION CalculateAge(@BirthDate DATE) RETURNS INT AS BEGIN RETURN DATEDIFF(YEAR, @BirthDate, GETDATE()); END;
