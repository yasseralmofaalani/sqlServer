دوال التحويل

تحويل أنظمة الإحداثيات
-- ST_Transform
SELECT ST_Transform(location, 3857) FROM cities;

-- ST_SetSRID
SELECT ST_SetSRID(ST_MakePoint(46.6753, 24.7136), 4326);

دوال التحليل

دوال التحليل
-- ST_Simplify
SELECT ST_Simplify(geometry, 0.001) FROM roads;

-- ST_ConvexHull
SELECT ST_ConvexHull(ST_Collect(location)) FROM cities;

-- ST_DelaunayTriangles
SELECT ST_DelaunayTriangles(ST_Collect(location)) FROM cities;
