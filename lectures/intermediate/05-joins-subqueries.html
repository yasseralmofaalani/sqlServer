الربط المكاني (Spatial Joins)

الربط المكاني
-- المدن في كل منطقة
SELECT r.name AS region, COUNT(c.id) AS city_count
FROM regions r
LEFT JOIN cities c ON ST_Within(c.location, r.boundary)
GROUP BY r.id, r.name;

-- الطرق في كل منطقة
SELECT r.name, SUM(ST_Length(ST_Transform(rd.geometry, 3857))) / 1000 AS total_km
FROM regions r
JOIN roads rd ON ST_Intersects(r.boundary, rd.geometry)
GROUP BY r.id, r.name;

الاستعلامات الفرعية المكانية

استعلامات فرعية
-- المدن الأقرب لكل منطقة
SELECT r.name,
    (SELECT c.name 
     FROM cities c 
     ORDER BY ST_Distance(c.location, r.centroid) 
     LIMIT 1) AS nearest_city
FROM regions r;
