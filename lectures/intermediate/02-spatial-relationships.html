المحاضرة 02: العلاقات المكانية

المستوى: المتوسط المدة: 90 دقيقة
← العودة إلى الفهرس

مقدمة

العلاقات المكانية هي الأساس في تحليل البيانات الجغرافية. PostGIS يوفر مجموعة شاملة من الدوال للتحقق من العلاقات المكانية بين الميزات الجغرافية.

الدوال الأساسية للعلاقات المكانية

ST_Intersects - التقاطع

التحقق من التقاطع
-- الطرق التي تتقاطع مع منطقة معينة
SELECT r.name, r.road_type
FROM roads r, regions reg
WHERE reg.name = 'منطقة الرياض'
  AND ST_Intersects(r.geometry, reg.boundary);

-- المناطق المتداخلة
SELECT r1.name AS region1, r2.name AS region2
FROM regions r1, regions r2
WHERE ST_Intersects(r1.boundary, r2.boundary)
  AND r1.id < r2.id;

ST_Contains - يحتوي

التحقق من الاحتواء
-- المدن داخل منطقة
SELECT c.name
FROM cities c, regions r
WHERE r.name = 'منطقة الرياض'
  AND ST_Contains(r.boundary, c.location);

-- المناطق التي تحتوي على مناطق أخرى
SELECT r1.name AS parent, r2.name AS child
FROM regions r1, regions r2
WHERE ST_Contains(r1.boundary, r2.boundary)
  AND r1.id != r2.id;

ST_Within - داخل

التحقق من الداخل
-- نفس ST_Contains لكن بالعكس
SELECT c.name
FROM cities c, regions r
WHERE r.name = 'منطقة الرياض'
  AND ST_Within(c.location, r.boundary);

ST_Touches - التلامس

التحقق من التلامس
-- المناطق المتجاورة
SELECT r1.name AS region1, r2.name AS region2
FROM regions r1, regions r2
WHERE ST_Touches(r1.boundary, r2.boundary)
  AND r1.id < r2.id;

ST_Overlaps - التداخل

التحقق من التداخل
-- المناطق المتداخلة جزئياً
SELECT r1.name AS region1, r2.name AS region2
FROM regions r1, regions r2
WHERE ST_Overlaps(r1.boundary, r2.boundary)
  AND r1.id < r2.id;

ST_Crosses - التقاطع

التحقق من التقاطع (للخطوط)
-- الطرق المتقاطعة
SELECT r1.name AS road1, r2.name AS road2
FROM roads r1, roads r2
WHERE ST_Crosses(r1.geometry, r2.geometry)
  AND r1.id < r2.id;

ST_Disjoint - منفصل

التحقق من الانفصال
-- المناطق المنفصلة
SELECT r1.name AS region1, r2.name AS region2
FROM regions r1, regions r2
WHERE ST_Disjoint(r1.boundary, r2.boundary)
  AND r1.id < r2.id;

ST_Equals - التساوي

التحقق من التساوي
-- الميزات المتطابقة
SELECT r1.name AS region1, r2.name AS region2
FROM regions r1, regions r2
WHERE ST_Equals(r1.boundary, r2.boundary)
  AND r1.id < r2.id;

الدوال المتقدمة

ST_DWithin - ضمن مسافة

البحث ضمن مسافة
-- المدن ضمن 50 كيلومتر من الرياض
SELECT c.name,
    ST_Distance(
        ST_Transform(riyadh.location, 3857),
        ST_Transform(c.location, 3857)
    ) / 1000 AS distance_km
FROM cities c,
     (SELECT location FROM cities WHERE name = 'الرياض') AS riyadh
WHERE ST_DWithin(
    ST_Transform(riyadh.location, 3857),
    ST_Transform(c.location, 3857),
    50000  -- 50 كيلومتر
)
AND c.name != 'الرياض';

ST_Relate - العلاقة التفصيلية

العلاقة التفصيلية (DE-9IM)
-- استخدام مصفوفة DE-9IM
SELECT 
    r1.name,
    r2.name,
    ST_Relate(r1.boundary, r2.boundary) AS relationship
FROM regions r1, regions r2
WHERE r1.id < r2.id
LIMIT 10;

مصفوفة العلاقات المكانية

الدالة الوصف مثال الاستخدام
ST_Intersects يتقاطع مع الطرق في منطقة
ST_Contains يحتوي على منطقة تحتوي مدينة
ST_Within داخل مدينة داخل منطقة
ST_Touches يلمس مناطق متجاورة
ST_Overlaps يتداخل مناطق متداخلة
ST_Crosses يتقاطع طرق متقاطعة
ST_Disjoint منفصل مناطق غير متجاورة
ST_Equals متساوي ميزات متطابقة

مثال شامل

تحليل العلاقات المكانية
-- إنشاء جدول للعلاقات المكانية
CREATE TABLE spatial_relationships AS
SELECT 
    r1.name AS region1,
    r2.name AS region2,
    CASE 
        WHEN ST_Contains(r1.boundary, r2.boundary) THEN 'contains'
        WHEN ST_Within(r1.boundary, r2.boundary) THEN 'within'
        WHEN ST_Touches(r1.boundary, r2.boundary) THEN 'touches'
        WHEN ST_Overlaps(r1.boundary, r2.boundary) THEN 'overlaps'
        WHEN ST_Intersects(r1.boundary, r2.boundary) THEN 'intersects'
        ELSE 'disjoint'
    END AS relationship_type,
    ST_Area(
        ST_Transform(
            ST_Intersection(r1.boundary, r2.boundary),
            3857
        )
    ) / 1000000 AS intersection_area_sqkm
FROM regions r1, regions r2
WHERE r1.id < r2.id
  AND ST_Intersects(r1.boundary, r2.boundary);
