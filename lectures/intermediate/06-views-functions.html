إنشاء العروض (Views)

عروض مكانية
-- عرض للمدن مع مناطقها
CREATE VIEW cities_with_regions AS
SELECT 
    c.id,
    c.name,
    c.population,
    r.name AS region_name,
    c.location
FROM cities c
JOIN regions r ON ST_Within(c.location, r.boundary);

-- عرض مادي (Materialized View)
CREATE MATERIALIZED VIEW region_statistics AS
SELECT 
    r.name,
    COUNT(c.id) AS city_count,
    SUM(c.population) AS total_population
FROM regions r
LEFT JOIN cities c ON ST_Within(c.location, r.boundary)
GROUP BY r.id, r.name;

الدوال المخصصة

دوال مخصصة مكانية
-- دالة لحساب المدن القريبة
CREATE OR REPLACE FUNCTION nearby_cities(
    city_name VARCHAR,
    distance_km NUMERIC
)
RETURNS TABLE(
    name VARCHAR,
    distance_km NUMERIC
) AS $$
BEGIN
    RETURN QUERY
    SELECT 
        c.name,
        ST_Distance(
            ST_Transform(c1.location, 3857),
            ST_Transform(c.location, 3857)
        ) / 1000 AS distance
    FROM cities c, cities c1
    WHERE c1.name = city_name
      AND ST_DWithin(
          ST_Transform(c1.location, 3857),
          ST_Transform(c.location, 3857),
          distance_km * 1000
      )
      AND c.name != city_name
    ORDER BY distance;
END;
$$ LANGUAGE plpgsql;
