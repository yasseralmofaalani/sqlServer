EXPLAIN و EXPLAIN ANALYZE
-- تحليل خطة التنفيذ
EXPLAIN SELECT * FROM cities WHERE ST_Within(location, 
    (SELECT boundary FROM regions WHERE name = 'منطقة الرياض'));

-- تحليل مع إحصائيات التنفيذ
EXPLAIN ANALYZE SELECT * FROM cities c, regions r
WHERE ST_Within(c.location, r.boundary);

تحسين الاستعلامات

  • استخدم الفهارس المكانية
  • استخدم ST_DWithin بدلاً من ST_Distance
  • حول الإحداثيات فقط عند الحاجة
  • استخدم VACUUM ANALYZE بانتظام
