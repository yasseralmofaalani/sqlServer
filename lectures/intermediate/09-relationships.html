المحاضرة 9: العلاقات بين الجداول (Relationships)

مقدم الدورة: م. ياسر المفعلاني

  • الصفحة الرئيسية
  • المحتوى

ما هي العلاقات؟

العلاقات تربط الجداول ببعضها البعض لضمان سلامة البيانات وإنشاء قاعدة بيانات منظمة.

المفتاح الأساسي (Primary Key)

المفتاح الأساسي هو عمود أو مجموعة أعمدة تحدد بشكل فريد كل سجل في الجدول.

إنشاء Primary Key

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, العمر INT ); -- أو بطريقة أخرى CREATE TABLE الطلاب ( رقم_الطالب INT, الاسم NVARCHAR(50) NOT NULL, العمر INT, CONSTRAINT PK_Students PRIMARY KEY (رقم_الطالب) );

Primary Key المركب

CREATE TABLE التسجيلات ( رقم_الطالب INT, رقم_المادة INT, الدرجة DECIMAL(5,2), PRIMARY KEY (رقم_الطالب, رقم_المادة) );

المفتاح الخارجي (Foreign Key)

المفتاح الخارجي يربط جدول بجدول آخر ويضمن سلامة البيانات المرجعية.

إنشاء Foreign Key

-- جدول الطلاب (الجدول الرئيسي) CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL ); -- جدول التسجيلات (الجدول المرتبط) CREATE TABLE التسجيلات ( رقم_التسجيل INT PRIMARY KEY, رقم_الطالب INT, رقم_المادة INT, الدرجة DECIMAL(5,2), FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) );

إضافة Foreign Key لجدول موجود

ALTER TABLE التسجيلات ADD CONSTRAINT FK_Student FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب);

أنواع العلاقات

1. علاقة واحد إلى واحد (One-to-One)

كل سجل في الجدول الأول يرتبط بسجل واحد فقط في الجدول الثاني.

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) ); CREATE TABLE بيانات_الشخصية ( رقم_الطالب INT PRIMARY KEY, العنوان NVARCHAR(200), الهاتف NVARCHAR(20), FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) );

2. علاقة واحد إلى كثير (One-to-Many)

سجل واحد في الجدول الأول يمكن أن يرتبط بعدة سجلات في الجدول الثاني.

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) ); CREATE TABLE التسجيلات ( رقم_التسجيل INT PRIMARY KEY, رقم_الطالب INT, رقم_المادة INT, FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ); -- طالب واحد يمكن أن يكون له عدة تسجيلات

3. علاقة كثير إلى كثير (Many-to-Many)

تتطلب جدول وسيط (Junction Table).

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) ); CREATE TABLE المواد ( رقم_المادة INT PRIMARY KEY, اسم_المادة NVARCHAR(50) ); -- جدول وسيط CREATE TABLE تسجيل_المواد ( رقم_الطالب INT, رقم_المادة INT, PRIMARY KEY (رقم_الطالب, رقم_المادة), FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب), FOREIGN KEY (رقم_المادة) REFERENCES المواد(رقم_المادة) );

CASCADE Actions

تحدد ما يحدث عند تحديث أو حذف السجل الرئيسي.

CASCADE DELETE

ALTER TABLE التسجيلات ADD CONSTRAINT FK_Student FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ON DELETE CASCADE; -- عند حذف طالب، سيتم حذف تسجيلاته تلقائياً

CASCADE UPDATE

ALTER TABLE التسجيلات ADD CONSTRAINT FK_Student FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ON UPDATE CASCADE; -- عند تحديث رقم_الطالب، سيتم تحديثه في التسجيلات تلقائياً

NO ACTION (الافتراضي)

ALTER TABLE التسجيلات ADD CONSTRAINT FK_Student FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ON DELETE NO ACTION ON UPDATE NO ACTION; -- لا يمكن حذف أو تحديث السجل الرئيسي إذا كان مرتبطاً

SET NULL

ALTER TABLE التسجيلات ADD CONSTRAINT FK_Student FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ON DELETE SET NULL; -- عند حذف الطالب، سيتم تعيين رقم_الطالب إلى NULL

عرض العلاقات

-- عرض جميع Foreign Keys SELECT fk.name AS ForeignKeyName, OBJECT_NAME(fk.parent_object_id) AS TableName, COL_NAME(fc.parent_object_id, fc.parent_column_id) AS ColumnName, OBJECT_NAME(fk.referenced_object_id) AS ReferencedTableName, COL_NAME(fc.referenced_object_id, fc.referenced_column_id) AS ReferencedColumnName FROM sys.foreign_keys AS fk INNER JOIN sys.foreign_key_columns AS fc ON fk.object_id = fc.constraint_object_id;

حذف Foreign Key

ALTER TABLE التسجيلات DROP CONSTRAINT FK_Student;

مثال شامل

-- إنشاء قاعدة بيانات للجامعة CREATE DATABASE الجامعة; USE الجامعة; -- جدول الكليات CREATE TABLE الكليات ( رقم_الكلية INT PRIMARY KEY IDENTITY(1,1), اسم_الكلية NVARCHAR(100) NOT NULL ); -- جدول الطلاب CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY IDENTITY(1,1), الاسم NVARCHAR(50) NOT NULL, رقم_الكلية INT, FOREIGN KEY (رقم_الكلية) REFERENCES الكليات(رقم_الكلية) ); -- جدول المواد CREATE TABLE المواد ( رقم_المادة INT PRIMARY KEY IDENTITY(1,1), اسم_المادة NVARCHAR(100) NOT NULL ); -- جدول التسجيلات (Many-to-Many) CREATE TABLE التسجيلات ( رقم_الطالب INT, رقم_المادة INT, الدرجة DECIMAL(5,2), PRIMARY KEY (رقم_الطالب, رقم_المادة), FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ON DELETE CASCADE, FOREIGN KEY (رقم_المادة) REFERENCES المواد(رقم_المادة) ON DELETE CASCADE );

أفضل الممارسات

  • استخدم Primary Key لكل جدول
  • استخدم Foreign Keys لضمان سلامة البيانات
  • خطط للعلاقات قبل إنشاء الجداول
  • استخدم CASCADE بحذر
  • استخدم أسماء واضحة للقيود (Constraints)
  • تأكد من وجود Indexes على Foreign Keys
نصيحة: إنشاء Index على Foreign Key يحسن أداء JOIN operations.

الخلاصة

  • Primary Key يحدد السجلات بشكل فريد
  • Foreign Key يربط الجداول ببعضها
  • هناك ثلاثة أنواع من العلاقات: One-to-One, One-to-Many, Many-to-Many
  • CASCADE Actions تحدد سلوك التحديث والحذف
  • العلاقات تضمن سلامة البيانات
