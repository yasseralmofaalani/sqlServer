نظرة شاملة على Geography Data Type

Geography Data Type في SQL Server 2008 يدعم البيانات الجغرافية على سطح الكرة الأرضية باستخدام نظام إحداثيات إهليلجي (Ellipsoidal). هذا النوع مثالي للتطبيقات التي تحتاج إلى حسابات دقيقة للمسافات والمساحات على سطح الأرض.

أنظمة الإحداثيات (Coordinate Reference Systems)

SRID - Spatial Reference Identifier

كل كائن Geography يجب أن يكون له SRID يحدد نظام الإحداثيات المستخدم. أشهر SRIDs:

SRID النظام الوصف
4326 WGS 84 النظام العالمي الأكثر استخداماً (GPS)
4269 NAD83 نظام أمريكا الشمالية
3857 Web Mercator يستخدم في Google Maps و Bing Maps
32636 UTM Zone 36N للمنطقة الشرقية من مصر

إنشاء Point جغرافي

-- إنشاء نقطة باستخدام WGS84 (SRID 4326) DECLARE @cairo GEOGRAPHY; SET @cairo = geography::Point(31.2357, 30.0444, 4326); -- المعاملات: خط الطول (Longitude)، خط العرض (Latitude)، SRID -- عرض الإحداثيات SELECT @cairo.Lat AS خط_العرض, @cairo.Long AS خط_الطول, @cairo.STSrid AS SRID;
ملاحظة مهمة: في Geography، المعامل الأول هو خط الطول (Longitude) والثاني هو خط العرض (Latitude). الترتيب عكسي في بعض الأنظمة الأخرى.

إنشاء LineString جغرافي

LineString يمثل خطاً يتكون من عدة نقاط متصلة.

-- إنشاء خط من القاهرة إلى الإسكندرية DECLARE @route GEOGRAPHY; SET @route = geography::STLineFromText( 'LINESTRING(31.2357 30.0444, 29.9158 31.2001)', 4326 ); -- حساب طول الخط بالكيلومتر SELECT @route.STLength() / 1000 AS الطول_بالكيلومتر;

إنشاء Polygon جغرافي

Polygon يمثل منطقة مغلقة. يجب أن تكون النقطة الأولى والأخيرة متطابقتين.

-- إنشاء مضلع يمثل منطقة في القاهرة DECLARE @area GEOGRAPHY; SET @area = geography::STPolyFromText( 'POLYGON((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0))', 4326 ); -- حساب المساحة بالمتر المربع SELECT @area.STArea() AS المساحة_بالمتر_المربع; -- حساب المساحة بالكيلومتر المربع SELECT @area.STArea() / 1000000 AS المساحة_بالكيلومتر_المربع;

حساب المسافة بين نقطتين

STDistance يحسب المسافة بين كائنين جغرافيين بالمتر.

-- المسافة بين القاهرة والإسكندرية DECLARE @cairo GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); DECLARE @alexandria GEOGRAPHY = geography::Point(29.9158, 31.2001, 4326); -- المسافة بالمتر SELECT @cairo.STDistance(@alexandria) AS المسافة_بالمتر; -- المسافة بالكيلومتر SELECT @cairo.STDistance(@alexandria) / 1000 AS المسافة_بالكيلومتر; -- المسافة بالميل SELECT @cairo.STDistance(@alexandria) / 1609.34 AS المسافة_بالميل;

إنشاء جدول للمواقع الجغرافية

CREATE TABLE المدن ( رقم_المدينة INT PRIMARY KEY IDENTITY(1,1), اسم_المدينة NVARCHAR(100) NOT NULL, الموقع GEOGRAPHY NOT NULL, عدد_السكان INT, تاريخ_التأسيس DATE ); -- إدراج بيانات INSERT INTO المدن (اسم_المدينة, الموقع, عدد_السكان) VALUES ('القاهرة', geography::Point(31.2357, 30.0444, 4326), 9500000), ('الإسكندرية', geography::Point(29.9158, 31.2001, 4326), 5200000), ('الجيزة', geography::Point(31.2089, 30.0131, 4326), 3200000), ('شبرا الخيمة', geography::Point(31.2406, 30.1285, 4326), 1200000), ('بورسعيد', geography::Point(32.3078, 31.2654, 4326), 750000);

البحث عن المدن القريبة

-- البحث عن جميع المدن في نطاق 100 كيلومتر من القاهرة DECLARE @myLocation GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); SELECT اسم_المدينة, الموقع.Lat AS خط_العرض, الموقع.Long AS خط_الطول, الموقع.STDistance(@myLocation) / 1000 AS المسافة_بالكيلومتر, عدد_السكان FROM المدن WHERE الموقع.STDistance(@myLocation) <= 100000 AND الموقع.STSrid = 4326 ORDER BY الموقع.STDistance(@myLocation);

التحقق من صحة البيانات الجغرافية

-- التحقق من صحة الكائن الجغرافي DECLARE @point GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); SELECT @point.STIsValid() AS صحيح, @point.STIsValidDetailed() AS تفاصيل_الصحة; -- إصلاح كائن غير صحيح DECLARE @invalid GEOGRAPHY; SET @invalid = geography::STGeomFromText('POINT(200 100)', 4326); -- هذا غير صحيح لأن خط الطول يجب أن يكون بين -180 و 180 IF @invalid.STIsValid() = 0 BEGIN SET @invalid = @invalid.MakeValid(); END;

تحويل SRID

يمكن تحويل البيانات من SRID إلى آخر، لكن SQL Server 2008 لا يدعم هذا مباشرة. يجب استخدام مكتبات خارجية أو تحديث SQL Server.

استخدام MultiPoint

-- إنشاء MultiPoint (عدة نقاط) DECLARE @points GEOGRAPHY; SET @points = geography::STGeomFromText( 'MULTIPOINT((31.2357 30.0444), (29.9158 31.2001), (31.2089 30.0131))', 4326 ); -- عدد النقاط SELECT @points.STNumGeometries() AS عدد_النقاط;

استخدام MultiPolygon

-- إنشاء MultiPolygon (عدة مناطق) DECLARE @regions GEOGRAPHY; SET @regions = geography::STGeomFromText( 'MULTIPOLYGON(((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0)), ((31.4 30.2, 31.5 30.2, 31.5 30.3, 31.4 30.3, 31.4 30.2)))', 4326 ); -- المساحة الإجمالية SELECT @regions.STArea() / 1000000 AS المساحة_بالكيلومتر_المربع;

أفضل الممارسات

  • استخدم SRID 4326 (WGS84) كمعيار عالمي
  • تأكد من صحة البيانات الجغرافية قبل الإدراج
  • استخدم Spatial Indexes لتحسين الأداء
  • تأكد من أن جميع الكائنات في نفس SRID عند إجراء العمليات
  • استخدم STDistance بحذر - قد يكون بطيئاً على البيانات الكبيرة
  • فكر في استخدام Geometry للبيانات المحلية الصغيرة
نصيحة: للمسافات القصيرة (أقل من 100 كم)، يمكن استخدام Geometry بدلاً من Geography للحصول على أداء أفضل.

الخلاصة

  • Geography مناسب للبيانات على سطح الكرة الأرضية
  • SRID يحدد نظام الإحداثيات
  • STDistance يحسب المسافات الدقيقة على سطح الأرض
  • STArea يحسب المساحات الدقيقة
  • تأكد من صحة البيانات قبل الاستخدام
