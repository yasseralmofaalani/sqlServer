إنشاء Trigger

CREATE TRIGGER TR_Students_Update ON الطلاب AFTER UPDATE AS BEGIN PRINT 'تم تحديث بيانات طالب'; END;

استخدام INSERTED و DELETED

CREATE TRIGGER TR_Students_Log ON الطلاب AFTER INSERT AS BEGIN INSERT INTO سجل_التغييرات (رقم_الطالب, العملية, التاريخ) SELECT رقم_الطالب, 'إدراج', GETDATE() FROM INSERTED; END;
