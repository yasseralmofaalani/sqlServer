مشروع نظام GIS متكامل

في هذا المشروع سنقوم بإنشاء نظام GIS كامل يربط SQL Server مع برامج GIS.

إنشاء قاعدة البيانات

CREATE DATABASE AdvancedGIS_System; USE AdvancedGIS_System; -- جدول المدن CREATE TABLE المدن ( ObjectID INT IDENTITY(1,1) PRIMARY KEY, اسم_المدينة NVARCHAR(100) NOT NULL, Shape GEOGRAPHY NOT NULL, عدد_السكان INT, تاريخ_التأسيس DATE ); -- جدول المناطق CREATE TABLE المناطق ( ObjectID INT IDENTITY(1,1) PRIMARY KEY, اسم_المنطقة NVARCHAR(100) NOT NULL, Shape GEOGRAPHY NOT NULL, المساحة DECIMAL(10,2) ); -- Spatial Indexes CREATE SPATIAL INDEX SIX_Cities ON المدن(Shape) USING GEOGRAPHY_GRID WITH ( GRIDS = (LEVEL_1 = MEDIUM, LEVEL_2 = MEDIUM, LEVEL_3 = MEDIUM, LEVEL_4 = MEDIUM), CELLS_PER_OBJECT = 16 ); CREATE SPATIAL INDEX SIX_Regions ON المناطق(Shape) USING GEOGRAPHY_GRID WITH ( GRIDS = (LEVEL_1 = MEDIUM, LEVEL_2 = MEDIUM, LEVEL_3 = MEDIUM, LEVEL_4 = MEDIUM), CELLS_PER_OBJECT = 16 );
