إنشاء مستخدم

CREATE LOGIN NewUser WITH PASSWORD = 'StrongPassword123!'; USE MyDatabase; CREATE USER NewUser FOR LOGIN NewUser;

منح الصلاحيات

GRANT SELECT, INSERT, UPDATE ON الطلاب TO NewUser; GRANT EXECUTE ON GetStudents TO NewUser;

الأدوار (Roles)

CREATE ROLE StudentManager; GRANT SELECT, INSERT, UPDATE ON الطلاب TO StudentManager; ALTER ROLE StudentManager ADD MEMBER NewUser;
