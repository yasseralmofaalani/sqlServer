إنشاء View

CREATE VIEW VW_Students_Details AS SELECT ط.رقم_الطالب, ط.الاسم, ت.رقم_المادة, ت.الدرجة FROM الطلاب ط INNER JOIN التسجيلات ت ON ط.رقم_الطالب = ت.رقم_الطالب;

استخدام View

SELECT * FROM VW_Students_Details;
