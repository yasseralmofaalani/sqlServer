المضلعات المعقدة

المضلعات المعقدة قد تحتوي على ثقوب (Holes) أو عدة أجزاء.

-- مضلع بثقب (Polygon with Hole) DECLARE @polygonWithHole GEOGRAPHY = geography::STGeomFromText( 'POLYGON((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0), (31.22 30.02, 31.22 30.08, 31.28 30.08, 31.28 30.02, 31.22 30.02))', 4326 ); -- الجزء الأول هو الحد الخارجي، الجزء الثاني هو الثقب

MultiPolygon

DECLARE @multiPolygon GEOGRAPHY = geography::STGeomFromText( 'MULTIPOLYGON(((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0)), ((31.4 30.2, 31.5 30.2, 31.5 30.3, 31.4 30.3, 31.4 30.2)))', 4326 ); -- عدد المضلعات SELECT @multiPolygon.STNumGeometries() AS عدد_المضلعات; -- الحصول على مضلع محدد SELECT @multiPolygon.STGeometryN(1) AS المضلع_الأول;

Union - دمج المناطق

DECLARE @region1 GEOGRAPHY = geography::STPolyFromText( 'POLYGON((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0))', 4326 ); DECLARE @region2 GEOGRAPHY = geography::STPolyFromText( 'POLYGON((31.25 30.05, 31.35 30.05, 31.35 30.15, 31.25 30.15, 31.25 30.05))', 4326 ); -- دمج المنطقتين DECLARE @union GEOGRAPHY = @region1.STUnion(@region2); SELECT @union.STArea() / 1000000 AS المساحة_المجمعة;

Intersection - التقاطع

-- المنطقة المشتركة بين منطقتين DECLARE @intersection GEOGRAPHY = @region1.STIntersection(@region2); SELECT @intersection.STArea() / 1000000 AS مساحة_التقاطع;

Difference - الفرق

-- المنطقة في region1 وليست في region2 DECLARE @difference GEOGRAPHY = @region1.STDifference(@region2); SELECT @difference.STArea() / 1000000 AS مساحة_الفرق;

SymDifference - الفرق المتماثل

-- المناطق في region1 أو region2 وليست في كليهما DECLARE @symDiff GEOGRAPHY = @region1.STSymDifference(@region2);

الخلاصة

  • المضلعات المعقدة قد تحتوي على ثقوب
  • MultiPolygon يحتوي على عدة مضلعات
  • STUnion لدمج المناطق
  • STIntersection للتقاطع
  • STDifference للفرق
