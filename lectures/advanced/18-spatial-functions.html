دوال المسافة والقياس

STDistance - حساب المسافة

DECLARE @point1 GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); DECLARE @point2 GEOGRAPHY = geography::Point(32.2995, 30.5877, 4326); -- المسافة بالمتر SELECT @point1.STDistance(@point2) AS المسافة_بالمتر; -- المسافة بالكيلومتر SELECT @point1.STDistance(@point2) / 1000 AS المسافة_بالكيلومتر;

STLength - حساب الطول

DECLARE @line GEOGRAPHY = geography::STLineFromText( 'LINESTRING(31.2357 30.0444, 29.9158 31.2001, 32.2995 30.5877)', 4326 ); SELECT @line.STLength() / 1000 AS الطول_بالكيلومتر;

STArea - حساب المساحة

DECLARE @polygon GEOGRAPHY = geography::STPolyFromText( 'POLYGON((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0))', 4326 ); SELECT @polygon.STArea() / 1000000 AS المساحة_بالكيلومتر_المربع;

دوال العلاقات المكانية

STIntersects - التحقق من التقاطع

DECLARE @point GEOGRAPHY = geography::Point(31.25, 30.05, 4326); DECLARE @polygon GEOGRAPHY = geography::STPolyFromText( 'POLYGON((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0))', 4326 ); SELECT @point.STIntersects(@polygon) AS يتقاطع; -- يعيد 1 إذا تقاطع، 0 إذا لم يتقاطع

STContains - التحقق من الاحتواء

SELECT @polygon.STContains(@point) AS يحتوي; -- يعيد 1 إذا كان المضلع يحتوي على النقطة

STWithin - التحقق من التواجد داخل

SELECT @point.STWithin(@polygon) AS داخل; -- يعيد 1 إذا كانت النقطة داخل المضلع

STTouches - التحقق من التلامس

DECLARE @polygon1 GEOGRAPHY = geography::STPolyFromText( 'POLYGON((31.2 30.0, 31.3 30.0, 31.3 30.1, 31.2 30.1, 31.2 30.0))', 4326 ); DECLARE @polygon2 GEOGRAPHY = geography::STPolyFromText( 'POLYGON((31.3 30.0, 31.4 30.0, 31.4 30.1, 31.3 30.1, 31.3 30.0))', 4326 ); SELECT @polygon1.STTouches(@polygon2) AS يتلامس;

STOverlaps - التحقق من التداخل

SELECT @polygon1.STOverlaps(@polygon2) AS يتداخل;

دوال التحويل والإنشاء

STBuffer - إنشاء منطقة عازلة

DECLARE @point GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); DECLARE @buffer GEOGRAPHY = @point.STBuffer(5000); -- منطقة عازلة بمسافة 5 كيلومتر (5000 متر) SELECT @buffer.STArea() / 1000000 AS مساحة_المنطقة_العازلة;

STUnion - دمج كائنين

DECLARE @union GEOGRAPHY = @polygon1.STUnion(@polygon2); SELECT @union.STArea() AS المساحة_المجمعة;

STIntersection - التقاطع

DECLARE @intersection GEOGRAPHY = @polygon1.STIntersection(@polygon2); SELECT @intersection.STArea() AS مساحة_التقاطع;

STDifference - الفرق

DECLARE @difference GEOGRAPHY = @polygon1.STDifference(@polygon2); SELECT @difference.STArea() AS مساحة_الفرق;

دوال المعلومات

STGeometryType - نوع الكائن

SELECT Shape.STGeometryType() AS نوع_الكائن, COUNT(*) AS العدد FROM المواقع GROUP BY Shape.STGeometryType();

STNumPoints - عدد النقاط

SELECT @line.STNumPoints() AS عدد_النقاط;

STPointN - الحصول على نقطة محددة

SELECT @line.STPointN(1) AS النقطة_الأولى;

مثال شامل - البحث عن المواقع القريبة

DECLARE @myLocation GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); DECLARE @searchRadius FLOAT = 10000; -- 10 كيلومتر SELECT اسم_الموقع, الموقع.STDistance(@myLocation) / 1000 AS المسافة_بالكيلومتر, الموقع.STBuffer(@searchRadius).STArea() / 1000000 AS مساحة_المنطقة_العازلة FROM المواقع WHERE الموقع.STDistance(@myLocation) <= @searchRadius AND الموقع.STIntersects(@myLocation.STBuffer(@searchRadius)) = 1 ORDER BY الموقع.STDistance(@myLocation);

الخلاصة

  • STDistance: حساب المسافة
  • STLength: حساب الطول
  • STArea: حساب المساحة
  • STIntersects, STContains, STWithin: علاقات مكانية
  • STBuffer: إنشاء منطقة عازلة
  • STUnion, STIntersection, STDifference: عمليات هندسية
