إنشاء قاعدة بيانات GIS

CREATE DATABASE GIS_System; USE GIS_System; -- جدول المدن CREATE TABLE المدن ( رقم_المدينة INT PRIMARY KEY IDENTITY(1,1), اسم_المدينة NVARCHAR(100) NOT NULL, الموقع GEOGRAPHY NOT NULL, عدد_السكان INT ); -- جدول المناطق CREATE TABLE المناطق ( رقم_المنطقة INT PRIMARY KEY IDENTITY(1,1), اسم_المنطقة NVARCHAR(100) NOT NULL, الحدود GEOGRAPHY NOT NULL, المساحة DECIMAL(10,2) ); -- إنشاء Spatial Index CREATE SPATIAL INDEX SIX_Cities_Location ON المدن(الموقع) USING GEOGRAPHY_GRID WITH ( GRIDS = (LEVEL_1 = MEDIUM, LEVEL_2 = MEDIUM, LEVEL_3 = MEDIUM, LEVEL_4 = MEDIUM), CELLS_PER_OBJECT = 16 );

إدراج بيانات جغرافية

INSERT INTO المدن (اسم_المدينة, الموقع, عدد_السكان) VALUES ('القاهرة', geography::Point(31.2357, 30.0444, 4326), 9500000), ('الإسكندرية', geography::Point(29.9158, 31.2001, 4326), 5200000), ('الجيزة', geography::Point(31.2089, 30.0131, 4326), 3200000);

استعلامات جغرافية

-- البحث عن المدن القريبة DECLARE @myLocation GEOGRAPHY = geography::Point(31.2357, 30.0444, 4326); SELECT اسم_المدينة, الموقع.STDistance(@myLocation) / 1000 AS المسافة_بالكيلومتر, عدد_السكان FROM المدن WHERE الموقع.STDistance(@myLocation) <= 100000 ORDER BY الموقع.STDistance(@myLocation);
