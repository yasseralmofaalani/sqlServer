Trigger للتحقق من صحة البيانات الجغرافية

CREATE TRIGGER TR_ValidateGeography ON المواقع AFTER INSERT, UPDATE AS BEGIN IF EXISTS ( SELECT 1 FROM INSERTED WHERE Shape.STIsValid() = 0 ) BEGIN RAISERROR('البيانات الجغرافية غير صحيحة', 16, 1); ROLLBACK TRANSACTION; END; END;

  • استخدم Triggers للتحقق من صحة البيانات
  • تأكد من SRID الصحيح
