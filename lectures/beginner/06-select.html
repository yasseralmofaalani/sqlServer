الاستعلام البسيط

SELECT هو الأمر الأساسي لاسترجاع البيانات من الجداول.

استرجاع جميع الأعمدة

SELECT * FROM الطلاب;

استرجاع أعمدة محددة

SELECT الاسم, العمر, التخصص FROM الطلاب;

WHERE - شرط التصفية

WHERE يستخدم لتصفية السجلات حسب شروط معينة.

مقارنات بسيطة

-- العمر يساوي 20 SELECT * FROM الطلاب WHERE العمر = 20; -- العمر أكبر من 20 SELECT * FROM الطلاب WHERE العمر > 20; -- العمر أكبر من أو يساوي 20 SELECT * FROM الطلاب WHERE العمر >= 20; -- العمر أقل من 25 SELECT * FROM الطلاب WHERE العمر < 25; -- العمر لا يساوي 20 SELECT * FROM الطلاب WHERE العمر <> 20;

AND و OR

-- العمر أكبر من 20 والتخصص هو 'علوم الحاسب' SELECT * FROM الطلاب WHERE العمر > 20 AND التخصص = 'علوم الحاسب'; -- العمر يساوي 20 أو 21 SELECT * FROM الطلاب WHERE العمر = 20 OR العمر = 21; -- العمر بين 20 و 25 SELECT * FROM الطلاب WHERE العمر >= 20 AND العمر <= 25;

BETWEEN

-- العمر بين 20 و 25 (شامل) SELECT * FROM الطلاب WHERE العمر BETWEEN 20 AND 25;

IN

-- التخصص في قائمة معينة SELECT * FROM الطلاب WHERE التخصص IN ('علوم الحاسب', 'الهندسة', 'الطب');

LIKE - البحث النمطي

-- الاسم يبدأ بـ 'أحمد' SELECT * FROM الطلاب WHERE الاسم LIKE 'أحمد%'; -- الاسم ينتهي بـ 'محمد' SELECT * FROM الطلاب WHERE الاسم LIKE '%محمد'; -- الاسم يحتوي على 'أحمد' SELECT * FROM الطلاب WHERE الاسم LIKE '%أحمد%'; -- الاسم يحتوي على حرف واحد قبل 'حمد' SELECT * FROM الطلاب WHERE الاسم LIKE '_حمد%';

NULL

-- العمر NULL SELECT * FROM الطلاب WHERE العمر IS NULL; -- العمر ليس NULL SELECT * FROM الطلاب WHERE العمر IS NOT NULL;

ORDER BY - الترتيب

-- ترتيب حسب العمر تصاعدياً SELECT * FROM الطلاب ORDER BY العمر ASC; -- ترتيب حسب العمر تنازلياً SELECT * FROM الطلاب ORDER BY العمر DESC; -- ترتيب حسب عدة أعمدة SELECT * FROM الطلاب ORDER BY التخصص ASC, العمر DESC;

SELECT DISTINCT - القيم الفريدة

-- عرض التخصصات الفريدة فقط SELECT DISTINCT التخصص FROM الطلاب;

TOP - عدد محدد من السجلات

-- أول 5 طلاب SELECT TOP 5 * FROM الطلاب ORDER BY رقم_الطالب; -- أول 10% من السجلات SELECT TOP 10 PERCENT * FROM الطلاب;

AS - الأسماء المستعارة

-- استخدام AS لتسمية الأعمدة SELECT الاسم AS اسم_الطالب, العمر AS عمر_الطالب, التخصص AS تخصص_الطالب FROM الطلاب;

الحسابات في SELECT

-- حساب عمود جديد SELECT الاسم, العمر, العمر + 5 AS العمر_بعد_5_سنوات FROM الطلاب;

مثال شامل

-- استعلام معقد SELECT TOP 10 رقم_الطالب, الاسم AS اسم_الطالب, العمر, التخصص FROM الطلاب WHERE العمر BETWEEN 20 AND 25 AND التخصص IN ('علوم الحاسب', 'الهندسة') AND الاسم LIKE 'أ%' ORDER BY العمر DESC, الاسم ASC;

أفضل الممارسات

  • تجنب استخدام SELECT * في الإنتاج، حدد الأعمدة المطلوبة
  • استخدم WHERE لتصفية البيانات قبل الترتيب
  • استخدم INDEXED columns في WHERE لتحسين الأداء
  • استخدم AS لتسمية واضحة للأعمدة
  • استخدم TOP لتحديد عدد السجلات المطلوبة
نصيحة: استخدم WHERE قبل ORDER BY لتحسين الأداء.

الخلاصة

  • SELECT يستخدم لاسترجاع البيانات
  • WHERE لتصفية السجلات
  • ORDER BY لترتيب النتائج
  • DISTINCT للحصول على قيم فريدة
  • TOP لتحديد عدد السجلات
