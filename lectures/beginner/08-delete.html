المحاضرة 8: حذف البيانات (DELETE)

حذف سجل واحد

DELETE يستخدم لحذف البيانات من الجداول.

حذف سجل محدد

DELETE FROM الطلاب WHERE رقم_الطالب = 1;

حذف عدة سجلات

-- حذف جميع الطلاب في تخصص معين DELETE FROM الطلاب WHERE التخصص = 'علوم الحاسب'; -- حذف باستخدام IN DELETE FROM الطلاب WHERE رقم_الطالب IN (1, 2, 3); -- حذف باستخدام BETWEEN DELETE FROM الطلاب WHERE العمر BETWEEN 20 AND 25;

حذف مع شروط معقدة

-- حذف باستخدام AND DELETE FROM الطلاب WHERE التخصص = 'علوم الحاسب' AND العمر > 25; -- حذف باستخدام OR DELETE FROM الطلاب WHERE التخصص = 'علوم الحاسب' OR التخصص = 'الهندسة'; -- حذف باستخدام LIKE DELETE FROM الطلاب WHERE الاسم LIKE 'أحمد%';

حذف من جدول مرتبط

-- حذف باستخدام JOIN DELETE ط FROM الطلاب ط INNER JOIN التسجيلات ت ON ط.رقم_الطالب = ت.رقم_الطالب WHERE ت.الدرجة < 50;

حذف مع OUTPUT

OUTPUT يسمح بعرض البيانات المحذوفة:

DELETE FROM الطلاب OUTPUT DELETED.* WHERE رقم_الطالب = 1;

التحقق قبل الحذف

-- عرض البيانات قبل الحذف SELECT * FROM الطلاب WHERE رقم_الطالب = 1; -- الحذف DELETE FROM الطلاب WHERE رقم_الطالب = 1; -- التحقق بعد الحذف SELECT * FROM الطلاب WHERE رقم_الطالب = 1;

TRUNCATE TABLE - حذف جميع السجلات

TRUNCATE TABLE يحذف جميع السجلات من الجدول بشكل أسرع من DELETE.

-- حذف جميع السجلات TRUNCATE TABLE الطلاب;

الفرق بين DELETE و TRUNCATE:

DELETE TRUNCATE
يمكن استخدام WHERE لا يمكن استخدام WHERE (يحذف الكل)
أبطأ أسرع
يسجل في Transaction Log يسجل الحد الأدنى في Transaction Log
يمكن التراجع عنه صعب التراجع عنه
يعيد تعيين IDENTITY لا يعيد تعيين IDENTITY
يعمل مع Foreign Keys لا يعمل إذا كان هناك Foreign Keys
تحذير: TRUNCATE TABLE لا يمكن التراجع عنه بسهولة! استخدمه بحذر.

معالجة الأخطاء

BEGIN TRY BEGIN TRANSACTION; DELETE FROM الطلاب WHERE رقم_الطالب = 1; COMMIT TRANSACTION; PRINT 'تم الحذف بنجاح'; END TRY BEGIN CATCH ROLLBACK TRANSACTION; PRINT 'حدث خطأ: ' + ERROR_MESSAGE(); END CATCH;

حذف مع Foreign Keys

إذا كان الجدول مرتبطاً بجداول أخرى، قد تحتاج لحذف السجلات المرتبطة أولاً:

-- حذف السجلات المرتبطة أولاً DELETE FROM التسجيلات WHERE رقم_الطالب = 1; -- ثم حذف السجل الرئيسي DELETE FROM الطلاب WHERE رقم_الطالب = 1;

CASCADE DELETE

يمكن إعداد Foreign Key لحذف السجلات المرتبطة تلقائياً:

-- إنشاء Foreign Key مع CASCADE DELETE ALTER TABLE التسجيلات ADD CONSTRAINT FK_Student FOREIGN KEY (رقم_الطالب) REFERENCES الطلاب(رقم_الطالب) ON DELETE CASCADE; -- الآن عند حذف طالب، سيتم حذف تسجيلاته تلقائياً
تحذير مهم: بدون WHERE، سيتم حذف جميع السجلات في الجدول! تأكد دائماً من استخدام WHERE.

مثال خاطئ (خطير!)

-- ⚠️ هذا سيحذف جميع السجلات! DELETE FROM الطلاب; -- بدون WHERE، سيتم حذف جميع الطلاب!

مثال شامل

-- حذف شامل مع شروط متعددة BEGIN TRANSACTION; -- التحقق من عدد السجلات التي سيتم حذفها SELECT COUNT(*) AS عدد_السجلات_للحذف FROM الطلاب WHERE العمر BETWEEN 20 AND 25 AND التخصص = 'علوم الحاسب'; -- الحذف DELETE FROM الطلاب WHERE العمر BETWEEN 20 AND 25 AND التخصص = 'علوم الحاسب'; -- إذا كان كل شيء صحيح، قم بالتنفيذ -- COMMIT TRANSACTION; -- إذا كان هناك خطأ، قم بالتراجع -- ROLLBACK TRANSACTION;

أفضل الممارسات

  • استخدم WHERE دائماً لتحديد السجلات المطلوب حذفها
  • تحقق من البيانات قبل الحذف باستخدام SELECT
  • استخدم TRANSACTIONS للحذف المهم
  • استخدم OUTPUT للتحقق من البيانات المحذوفة
  • احتفظ بنسخة احتياطية قبل الحذف
  • تأكد من حذف السجلات المرتبطة أولاً (أو استخدم CASCADE)
  • استخدم TRY...CATCH لمعالجة الأخطاء
نصيحة: استخدم SELECT أولاً لمعرفة عدد السجلات التي سيتم حذفها قبل تنفيذ DELETE.

الخلاصة

  • DELETE يستخدم لحذف البيانات من الجداول
  • استخدم WHERE دائماً لتحديد السجلات
  • TRUNCATE TABLE أسرع لحذف جميع السجلات
  • استخدم TRANSACTIONS للحذف المهم
  • تأكد من حذف السجلات المرتبطة
