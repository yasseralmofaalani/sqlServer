المحاضرة 5: إدراج البيانات (INSERT)

مقدم الدورة: م. ياسر المفعلاني

  • الصفحة الرئيسية
  • المحتوى

إدراج سجل واحد

أبسط طريقة لإدراج البيانات هي استخدام INSERT INTO مع VALUES.

إدراج جميع الأعمدة

INSERT INTO الطلاب VALUES (1, 'أحمد محمد', 20, 'علوم الحاسب');

إدراج أعمدة محددة

INSERT INTO الطلاب (الاسم, العمر, التخصص) VALUES ('فاطمة علي', 21, 'الهندسة'); -- رقم_الطالب سيأخذ القيمة التلقائية إذا كان IDENTITY

إدراج عدة سجلات

INSERT INTO الطلاب (الاسم, العمر, التخصص) VALUES ('محمد خالد', 22, 'الطب'), ('سارة أحمد', 19, 'الصيدلة'), ('خالد يوسف', 23, 'الهندسة');

إدراج من جدول آخر

-- إنشاء جدول للطلاب الجدد CREATE TABLE الطلاب_الجدد ( الاسم NVARCHAR(50), العمر INT, التخصص NVARCHAR(50) ); -- إدراج البيانات من جدول آخر INSERT INTO الطلاب (الاسم, العمر, التخصص) SELECT الاسم, العمر, التخصص FROM الطلاب_الجدد;

إدراج مع SELECT

-- إدراج بيانات محسوبة INSERT INTO الطلاب (الاسم, العمر, التخصص) SELECT 'طالب جديد', 20, 'علوم الحاسب' WHERE NOT EXISTS ( SELECT 1 FROM الطلاب WHERE الاسم = 'طالب جديد' );

إدراج مع DEFAULT

-- استخدام القيم الافتراضية INSERT INTO الموظفون (الاسم_الكامل, البريد_الإلكتروني) VALUES ('أحمد محمد', 'ahmed@example.com'); -- تاريخ_التعيين و القسم سيأخذان القيم الافتراضية -- استخدام DEFAULT صراحة INSERT INTO الموظفون (الاسم_الكامل, البريد_الإلكتروني, تاريخ_التعيين) VALUES ('فاطمة علي', 'fatima@example.com', DEFAULT);

إدراج مع NULL

INSERT INTO الطلاب (الاسم, العمر, التخصص) VALUES ('محمد', NULL, 'الهندسة'); -- العمر سيكون NULL

مثال شامل

-- إنشاء جدول الطلاب CREATE TABLE الطلاب ( رقم_الطالب INT IDENTITY(1,1) PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, العمر INT, التخصص NVARCHAR(50), تاريخ_التسجيل DATETIME DEFAULT GETDATE() ); -- إدراج عدة طلاب INSERT INTO الطلاب (الاسم, العمر, التخصص) VALUES ('أحمد محمد علي', 20, 'علوم الحاسب'), ('فاطمة خالد', 21, 'الهندسة المدنية'), ('محمد يوسف', 19, 'الطب'), ('سارة أحمد', 22, 'الصيدلة'), ('خالد إبراهيم', 23, 'الهندسة الكهربائية');

التحقق من البيانات المدرجة

-- عرض جميع الطلاب SELECT * FROM الطلاب; -- عرض عدد السجلات SELECT COUNT(*) AS عدد_الطلاب FROM الطلاب;

معالجة الأخطاء

-- استخدام TRY...CATCH BEGIN TRY INSERT INTO الطلاب (الاسم, العمر, التخصص) VALUES ('أحمد', 20, 'علوم الحاسب'); PRINT 'تم الإدراج بنجاح'; END TRY BEGIN CATCH PRINT 'حدث خطأ: ' + ERROR_MESSAGE(); END CATCH;

إدراج مع OUTPUT

OUTPUT يسمح بعرض البيانات المدرجة:

INSERT INTO الطلاب (الاسم, العمر, التخصص) OUTPUT INSERTED.رقم_الطالب, INSERTED.الاسم VALUES ('علي حسن', 24, 'الهندسة');

أفضل الممارسات

  • حدد الأعمدة صراحة عند الإدراج لتجنب الأخطاء
  • استخدم المعاملات (Parameters) في التطبيقات لتجنب SQL Injection
  • تحقق من صحة البيانات قبل الإدراج
  • استخدم TRY...CATCH لمعالجة الأخطاء
  • استخدم OUTPUT للتحقق من البيانات المدرجة
تحذير: تأكد من أن البيانات المدرجة تتوافق مع قيود الجدول (Constraints).

الخلاصة

  • INSERT INTO يستخدم لإدراج البيانات في الجداول
  • يمكن إدراج سجل واحد أو عدة سجلات
  • يمكن إدراج البيانات من جدول آخر باستخدام SELECT
  • استخدم DEFAULT للقيم الافتراضية
  • استخدم TRY...CATCH لمعالجة الأخطاء
