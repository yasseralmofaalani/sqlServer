المحاضرة 7: تحديث البيانات (UPDATE)

مقدم الدورة: م. ياسر المفعلاني

  • الصفحة الرئيسية
  • المحتوى

تحديث سجل واحد

UPDATE يستخدم لتعديل البيانات الموجودة في الجداول.

تحديث عمود واحد

UPDATE الطلاب SET العمر = 21 WHERE رقم_الطالب = 1;

تحديث عدة أعمدة

UPDATE الطلاب SET العمر = 22, التخصص = 'الهندسة' WHERE رقم_الطالب = 1;

تحديث عدة سجلات

-- تحديث جميع الطلاب في تخصص معين UPDATE الطلاب SET العمر = العمر + 1 WHERE التخصص = 'علوم الحاسب';

تحديث مع شروط معقدة

-- تحديث باستخدام AND UPDATE الطلاب SET التخصص = 'علوم الحاسب المتقدم' WHERE التخصص = 'علوم الحاسب' AND العمر > 20; -- تحديث باستخدام IN UPDATE الطلاب SET العمر = العمر + 1 WHERE رقم_الطالب IN (1, 2, 3, 4, 5); -- تحديث باستخدام BETWEEN UPDATE الطلاب SET التخصص = 'تخصص محدث' WHERE العمر BETWEEN 20 AND 25;

تحديث مع الحسابات

-- زيادة العمر بمقدار 1 UPDATE الطلاب SET العمر = العمر + 1; -- ضرب الراتب في 1.1 (زيادة 10%) UPDATE الموظفون SET الراتب = الراتب * 1.1 WHERE القسم = 'المبيعات';

تحديث مع NULL

-- تعيين قيمة NULL UPDATE الطلاب SET التخصص = NULL WHERE رقم_الطالب = 1;

تحديث من جدول آخر

-- إنشاء جدول للتحديثات CREATE TABLE تحديثات_الطلاب ( رقم_الطالب INT, العمر_الجديد INT ); -- تحديث من جدول آخر UPDATE الطلاب SET العمر = تحديثات_الطلاب.العمر_الجديد FROM تحديثات_الطلاب WHERE الطلاب.رقم_الطالب = تحديثات_الطلاب.رقم_الطالب;

تحديث مع JOIN

-- تحديث مع INNER JOIN UPDATE ط SET ط.العمر = م.العمر_الجديد FROM الطلاب ط INNER JOIN تحديثات_الطلاب م ON ط.رقم_الطالب = م.رقم_الطالب;

تحديث مع OUTPUT

OUTPUT يسمح بعرض البيانات المحدثة:

UPDATE الطلاب SET العمر = العمر + 1 OUTPUT INSERTED.رقم_الطالب, DELETED.العمر AS العمر_القديم, INSERTED.العمر AS العمر_الجديد WHERE رقم_الطالب = 1;

التحقق قبل التحديث

-- عرض البيانات قبل التحديث SELECT * FROM الطلاب WHERE رقم_الطالب = 1; -- التحديث UPDATE الطلاب SET العمر = 22 WHERE رقم_الطالب = 1; -- التحقق بعد التحديث SELECT * FROM الطلاب WHERE رقم_الطالب = 1;

معالجة الأخطاء

BEGIN TRY BEGIN TRANSACTION; UPDATE الطلاب SET العمر = 22 WHERE رقم_الطالب = 1; COMMIT TRANSACTION; PRINT 'تم التحديث بنجاح'; END TRY BEGIN CATCH ROLLBACK TRANSACTION; PRINT 'حدث خطأ: ' + ERROR_MESSAGE(); END CATCH;
تحذير مهم: بدون WHERE، سيتم تحديث جميع السجلات في الجدول! تأكد دائماً من استخدام WHERE.

مثال خاطئ (خطير!)

-- ⚠️ هذا سيحدث جميع السجلات! UPDATE الطلاب SET العمر = 20; -- بدون WHERE، سيتم تحديث جميع الطلاب!

مثال شامل

-- تحديث شامل مع شروط متعددة UPDATE الطلاب SET العمر = العمر + 1, التخصص = 'تخصص محدث' WHERE العمر BETWEEN 20 AND 25 AND التخصص IN ('علوم الحاسب', 'الهندسة') AND الاسم LIKE 'أ%';

أفضل الممارسات

  • استخدم WHERE دائماً لتحديد السجلات المطلوب تحديثها
  • تحقق من البيانات قبل التحديث باستخدام SELECT
  • استخدم TRANSACTIONS للتحديثات المهمة
  • استخدم OUTPUT للتحقق من التحديثات
  • استخدم TRY...CATCH لمعالجة الأخطاء
  • احتفظ بنسخة احتياطية قبل التحديثات الكبيرة
نصيحة: استخدم SELECT أولاً لمعرفة عدد السجلات التي سيتم تحديثها قبل تنفيذ UPDATE.

الخلاصة

  • UPDATE يستخدم لتعديل البيانات الموجودة
  • استخدم WHERE دائماً لتحديد السجلات
  • يمكن تحديث عمود واحد أو عدة أعمدة
  • يمكن استخدام الحسابات في SET
  • استخدم TRANSACTIONS للتحديثات المهمة
