ما هو الجدول (Table)؟

الجدول هو الكائن الأساسي في قاعدة البيانات الذي يحتوي على البيانات. يتكون من صفوف (Rows) وأعمدة (Columns).

إنشاء جدول بسيط

USE MyDatabase; GO CREATE TABLE الطلاب ( رقم_الطالب INT, الاسم NVARCHAR(50), العمر INT, التخصص NVARCHAR(50) );

أنواع البيانات الشائعة

نوع البيانات الحجم الوصف
INT 4 bytes أرقام صحيحة من -2,147,483,648 إلى 2,147,483,647
BIGINT 8 bytes أرقام صحيحة كبيرة
SMALLINT 2 bytes أرقام صحيحة صغيرة
TINYINT 1 byte أرقام من 0 إلى 255
DECIMAL(p,s) 5-17 bytes أرقام عشرية دقيقة (p = الدقة، s = المقياس)
FLOAT 4-8 bytes أرقام عشرية تقريبية
VARCHAR(n) متغير نص متغير الطول (غير Unicode)
NVARCHAR(n) متغير نص Unicode متغير الطول (يدعم العربية)
CHAR(n) ثابت نص ثابت الطول
NCHAR(n) ثابت نص Unicode ثابت الطول
DATE 3 bytes تاريخ فقط (من 0001-01-01 إلى 9999-12-31)
TIME 3-5 bytes وقت فقط
DATETIME 8 bytes تاريخ ووقت
BIT 1 bit قيمة منطقية (0 أو 1)
نصيحة: استخدم NVARCHAR بدلاً من VARCHAR للدعم الكامل للغة العربية.

إنشاء جدول مع قيود (Constraints)

المفتاح الأساسي (Primary Key)

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, العمر INT, التخصص NVARCHAR(50) );

المفتاح الأساسي المركب

CREATE TABLE التسجيلات ( رقم_الطالب INT, رقم_المادة INT, الدرجة DECIMAL(5,2), PRIMARY KEY (رقم_الطالب, رقم_المادة) );

NOT NULL - عدم السماح بالقيم الفارغة

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, البريد_الإلكتروني NVARCHAR(100) NOT NULL, الهاتف NVARCHAR(20) NULL );

UNIQUE - القيم الفريدة

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, البريد_الإلكتروني NVARCHAR(100) UNIQUE, رقم_الهوية NVARCHAR(20) UNIQUE );

CHECK - التحقق من القيم

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, العمر INT CHECK (العمر >= 18 AND العمر <= 100), الدرجة DECIMAL(5,2) CHECK (الدرجة >= 0 AND الدرجة <= 100) );

DEFAULT - القيمة الافتراضية

CREATE TABLE الطلاب ( رقم_الطالب INT PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, تاريخ_التسجيل DATETIME DEFAULT GETDATE(), الحالة NVARCHAR(20) DEFAULT 'نشط' );

IDENTITY - الترقيم التلقائي

CREATE TABLE الطلاب ( رقم_الطالب INT IDENTITY(1,1) PRIMARY KEY, الاسم NVARCHAR(50) NOT NULL, العمر INT ); -- IDENTITY(1,1) يعني يبدأ من 1 ويزيد بمقدار 1

مثال شامل لإنشاء جدول

CREATE TABLE الموظفون ( رقم_الموظف INT IDENTITY(1,1) PRIMARY KEY, الاسم_الكامل NVARCHAR(100) NOT NULL, البريد_الإلكتروني NVARCHAR(100) UNIQUE NOT NULL, الهاتف NVARCHAR(20), الراتب DECIMAL(10,2) CHECK (الراتب > 0), تاريخ_التعيين DATE DEFAULT GETDATE(), القسم NVARCHAR(50) DEFAULT 'غير محدد', نشط BIT DEFAULT 1 );

عرض الجداول

-- عرض جميع الجداول في قاعدة البيانات SELECT TABLE_NAME FROM INFORMATION_SCHEMA.TABLES WHERE TABLE_TYPE = 'BASE TABLE'; -- عرض بنية جدول معين EXEC sp_help 'الطلاب'; -- عرض أعمدة جدول SELECT COLUMN_NAME, DATA_TYPE, IS_NULLABLE, COLUMN_DEFAULT FROM INFORMATION_SCHEMA.COLUMNS WHERE TABLE_NAME = 'الطلاب';

تعديل الجدول (ALTER TABLE)

إضافة عمود جديد

ALTER TABLE الطلاب ADD العنوان NVARCHAR(200);

تعديل عمود موجود

-- تغيير نوع البيانات أو الحجم ALTER TABLE الطلاب ALTER COLUMN الاسم NVARCHAR(100);

حذف عمود

ALTER TABLE الطلاب DROP COLUMN العنوان;

إضافة قيد

ALTER TABLE الطلاب ADD CONSTRAINT CHK_Age CHECK (العمر >= 18);

حذف قيد

ALTER TABLE الطلاب DROP CONSTRAINT CHK_Age;

حذف الجدول

-- حذف جدول DROP TABLE الطلاب; -- حذف جدول مع التحقق من وجوده IF OBJECT_ID('الطلاب', 'U') IS NOT NULL DROP TABLE الطلاب;
تحذير: حذف الجدول يحذف جميع البيانات الموجودة فيه ولا يمكن التراجع عنه!

أفضل الممارسات

  • استخدم أسماء واضحة ووصفية للجداول والأعمدة
  • استخدم NVARCHAR للدعم الكامل للعربية
  • حدد PRIMARY KEY لكل جدول
  • استخدم NOT NULL للأعمدة المطلوبة
  • استخدم CHECK للتحقق من صحة البيانات
  • استخدم IDENTITY للترقيم التلقائي
  • خطط لبنية الجدول قبل إنشائه

الخلاصة

  • الجدول هو الكائن الأساسي لتخزين البيانات
  • يمكن إضافة قيود مختلفة لضمان سلامة البيانات
  • يمكن تعديل الجدول باستخدام ALTER TABLE
  • التخطيط الجيد مهم قبل إنشاء الجداول
